22  package org.displaytag.model;
23  
24  import java.util.Iterator;
25  import java.util.List;
26  
27  import org.displaytag.decorator.TableDecorator;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  
32  /**
33   * Iterator on table rows.
34   * @author Fabrizio Giustina
35   * @version $Revision$ ($Author$)
36   */
37  public class RowIterator
38  {
39  
40      /**
41       * logger.
42       */
43      private static Logger log = LoggerFactory.getLogger(RowIterator.class);
44  
45      /**
46       * internal iterator for Rows.
47       */
48      private Iterator<Row> iterator;
49  
50      /**
51       * row number counter.
52       */
53      private int rowNumber;
54  
55      /**
56       * reference to the table TableDecorator.
57       */
58      private TableDecorator decorator;
59  
60      /**
61       * id inherited from the TableTag (needed only for logging).
62       */
63      private String id;
64  
65      /**
66       * Starting offset for items n the current page. Needed to calculare the index in the original list
67       */
68      private int pageOffset;
69  
70      /**
71       * Constructor for RowIterator.
72       * @param rowList List containing Row objects
73       * @param columnList List containing CellHeader objects
74       * @param tableDecorator TableDecorator
75       * @param offset Starting offset for items n the current page
76       */
77      protected RowIterator(List<Row> rowList, List<HeaderCell> columnList, TableDecorator tableDecorator, int offset)
78      {
79          this.iterator = rowList.iterator();
80          this.rowNumber = 0;
81          this.decorator = tableDecorator;
82          this.pageOffset = offset;
83      }
84  
85      /**
86       * Setter for the tablemodel id.
87       * @param tableId same id of table tag, needed for logging
88       */
89      public void setId(String tableId)
90      {
91          this.id = tableId;
92      }
93  
94      /**
95       * Check if a next row exist.
96       * @return boolean true if a new row
97       */
98      public boolean hasNext()
99      {
100         return this.iterator.hasNext();
101     }
102 
103     /**
104      * Gets the page offset.
105      *
106      * @return the page offset
107      */
108     public int getPageOffset()
109     {
110         return this.pageOffset;
111     }
112 
113     /**
114      * Returns the next row object.
115      * @return Row
116      */
117     public Row next()
118     {
119 
120         int currentRowNumber = this.rowNumber++;
121 
122         if (log.isDebugEnabled())
123         {
124             log.debug("[{}] rowIterator.next() row number=", this.id, currentRowNumber);
125         }
126 
127         Row row = this.iterator.next();
128 
129         row.setRowNumber(currentRowNumber);
130 
131         if (this.decorator != null)
132         {
133             this.decorator.initRow(row.getObject(), currentRowNumber, currentRowNumber + getPageOffset());
134         }
135 
136         return row;
137 
138     }
139 
140 }
