22  package org.displaytag.model;
23  
24  import java.util.Comparator;
25  
26  import org.apache.commons.lang3.builder.EqualsBuilder;
27  import org.apache.commons.lang3.builder.HashCodeBuilder;
28  import org.displaytag.decorator.TableDecorator;
29  import org.displaytag.exception.ObjectLookupException;
30  import org.displaytag.exception.RuntimeLookupException;
31  import org.displaytag.util.LookupUtil;
32  
33  
34  /**
35   * Comparator for rows.
36   * @author Fabrizio Giustina
37   * @version $Revision$ ($Author$)
38   */
39  public class RowSorter implements Comparator<Object>
40  {
41  
42      /**
43       * name of the property in bean.
44       */
45      private String property;
46  
47      /**
48       * table decorator.
49       */
50      private TableDecorator decorator;
51  
52      /**
53       * sort order ascending?
54       */
55      private boolean ascending;
56  
57      /**
58       * Index of the sorted column.
59       */
60      private int columnIndex;
61  
62      /**
63       * Comparator used for comparisons.
64       */
65      private Comparator<Object> comparator;
66  
67      /**
68       * Initialize a new RowSorter.
69       * @param sortedColumnIndex index of the sorted column
70       * @param beanProperty name of the property. If pProperty is null column index is used to get a static cell value
71       * from the row object
72       * @param tableDecorator TableDecorator instance
73       * @param ascendingOrder boolean ascending order?
74       * @param compar the comparator to use
75       */
76      public RowSorter(
77          int sortedColumnIndex,
78          String beanProperty,
79          TableDecorator tableDecorator,
80          boolean ascendingOrder,
81          Comparator<Object> compar)
82      {
83          this.columnIndex = sortedColumnIndex;
84          this.property = beanProperty;
85          this.decorator = tableDecorator;
86          this.ascending = ascendingOrder;
87          this.comparator = compar;
88          if (compar == null)
89          {
90              throw new IllegalArgumentException(
91                  "A null comparator has been passed to RowSorter. A comparator instance is required");
92          }
93      }
94  
95      /**
96       * Compares two objects by first fetching a property from each object and then comparing that value. If there are
97       * any errors produced while trying to compare these objects then a RunTimeException will be thrown as any error
98       * found here will most likely be a programming error that needs to be quickly addressed (like trying to compare
99       * objects that are not comparable, or trying to read a property from a bean that is invalid, etc...)
100      * @param object1 Object
101      * @param object2 Object
102      * @return int
103      * @see java.util.Comparator#compare(Object, Object)
104      */
105     @Override
106     public final int compare(Object object1, Object object2)
107     {
108 
109         Object obj1 = null;
110         Object obj2 = null;
111 
112         // if property is null compare using two static cell objects
113         if (this.property == null)
114         {
115             if (object1 instanceof Row)
116             {
117                 obj1 = ((Row) object1).getCellList().get(this.columnIndex);
118             }
119             if (object2 instanceof Row)
120             {
121                 obj2 = ((Row) object2).getCellList().get(this.columnIndex);
122             }
123 
124             return checkNullsAndCompare(obj1, obj2);
125         }
126 
127         if (object1 instanceof Row)
128         {
129             obj1 = ((Row) object1).getObject();
130         }
131         if (object2 instanceof Row)
132         {
133             obj2 = ((Row) object2).getObject();
134         }
135 
136         try
137         {
138             Object result1;
139             Object result2;
140 
141             // If they have supplied a decorator, then make sure and use it for the sorting as well
142             if (this.decorator != null && this.decorator.hasGetterFor(this.property))
143             {
144                 // set the row before sending to the decorator
145                 this.decorator.initRow(obj1, 0, 0);
146 
147                 result1 = LookupUtil.getBeanProperty(this.decorator, this.property);
148 
149                 // set the row before sending to the decorator
150                 this.decorator.initRow(obj2, 0, 0);
151 
152                 result2 = LookupUtil.getBeanProperty(this.decorator, this.property);
153             }
154             else
155             {
156                 result1 = LookupUtil.getBeanProperty(obj1, this.property);
157                 result2 = LookupUtil.getBeanProperty(obj2, this.property);
158             }
159 
160             return checkNullsAndCompare(result1, result2);
161         }
162         catch (ObjectLookupException e)
163         {
164             throw new RuntimeLookupException(getClass(), this.property, e);
165         }
166     }
167 
168     /**
169      * Compares two given objects, and handles the case where nulls are present.
170      * @param object1 first object to compare
171      * @param object2 second object to compare
172      * @return int result
173      */
174     private int checkNullsAndCompare(Object object1, Object object2)
175     {
176         int returnValue;
177         if (object1 == null && object2 != null)
178         {
179             returnValue = -1;
180         }
181         else if (object1 != null && object2 == null)
182         {
183             returnValue = 1;
184         }
185         else if (object1 == null && object2 == null)
186         {
187             // both null
188             returnValue = 0;
189         }
190         else
191         {
192             returnValue = this.comparator.compare(object1, object2);
193         }
194         int ascendingInt = this.ascending ? 1 : -1;
195         return ascendingInt * returnValue;
196     }
197 
198     /**
199      * Is this Comparator the same as another one?
200      * @param object Object
201      * @return boolean
202      * @see java.util.Comparator#equals(Object)
203      */
204     @Override
205     public final boolean equals(Object object)
206     {
207         if (object instanceof RowSorter)
208         {
209             return new EqualsBuilder()
210                 .append(this.property, ((RowSorter) object).property)
211                 .append(this.columnIndex, ((RowSorter) object).columnIndex)
212                 .isEquals();
213         }
214 
215         return false;
216     }
217 
218     /**
219      * @see java.lang.Object#hashCode()
220      */
221     @Override
222     public final int hashCode()
223     {
224         return new HashCodeBuilder(31, 33).append(this.property).append(this.columnIndex).toHashCode();
225     }
