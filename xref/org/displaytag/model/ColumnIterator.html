7   package org.displaytag.model;
8   
9   import java.util.Iterator;
10  import java.util.List;
11  
12  /**
13   * Iterator on columns.
14   */
15  public class ColumnIterator {
16  
17      /**
18       * current row.
19       */
20      private final Row parentRow;
21  
22      /**
23       * Internal iterator on header cells.
24       */
25      private final Iterator<HeaderCell> headerIterator;
26  
27      /**
28       * Internal iterator on cells.
29       */
30      private final Iterator<Cell> cellIterator;
31  
32      /**
33       * Creates a new ColumnIterator given a list of column and a row.
34       *
35       * @param columns
36       *            List containing column objects
37       * @param row
38       *            current Row
39       */
40      public ColumnIterator(final List<HeaderCell> columns, final Row row) {
41          this.headerIterator = columns.iterator();
42          this.cellIterator = row.getCellList().iterator();
43          this.parentRow = row;
44      }
45  
46      /**
47       * Are there more columns?.
48       *
49       * @return boolean <code>true</code> if there are more columns
50       */
51      public boolean hasNext() {
52          return this.headerIterator.hasNext();
53      }
54  
55      /**
56       * Returns the next column.
57       *
58       * @return Column next column
59       */
60      public Column nextColumn() {
61          final HeaderCell header = this.headerIterator.next();
62  
63          Cell cell = Cell.EMPTY_CELL;
64  
65          // if cells is not present simply return an empty cell.
66          // this is needed for automatic properties discovery
67          if (this.cellIterator.hasNext()) {
68              cell = this.cellIterator.next();
69          }
70  
71          // create a new column using the next value in the header and cell iterators and returns it
72          return new Column(header, cell, this.parentRow);
73      }
74  
75  }
