7   package org.displaytag.decorator;
8   
9   import jakarta.servlet.jsp.PageContext;
10  
11  import org.displaytag.exception.DecoratorException;
12  import org.displaytag.properties.MediaTypeEnum;
13  
14  /**
15   * Interface for simple column decorators.
16   * <p>
17   * A column decorator is called after the object has been retrieved and it can "transform" the object before the
18   * rendering.
19   * <p>
20   * The <code>DisplaytagColumnDecorator</code> interface has been introduced in displaytag 1.1 and replaces the previous
21   * <code>ColumnDecorator</code> interface, adding the pageContext and media parameters, and changing the return type to
22   * object to allow decorator chaining.
23   *
24   * @since 1.1
25   */
26  public interface DisplaytagColumnDecorator {
27  
28      /**
29       * Called after the object has been retrieved from the bean contained in the list. The decorate method is
30       * responsible for transforming the object into a string to render in the page.
31       *
32       * @param columnValue
33       *            Object to decorate
34       * @param pageContext
35       *            jsp page context
36       * @param media
37       *            current media (html, pdf, excel...)
38       *
39       * @return Object decorated object
40       *
41       * @throws DecoratorException
42       *             wrapper exception for any exception thrown during decoration
43       */
44      Object decorate(Object columnValue, PageContext pageContext, MediaTypeEnum media) throws DecoratorException;
45  
46  }
