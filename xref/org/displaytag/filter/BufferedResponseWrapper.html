7   package org.displaytag.filter;
8   
9   import jakarta.servlet.http.HttpServletResponse;
10  
11  /**
12   * Buffers the response; will not send anything directly through to the actual response. Note that this blocks the
13   * content-type from being set, you must set it manually in the response.
14   */
15  public interface BufferedResponseWrapper extends HttpServletResponse {
16  
17      /**
18       * Headers which cause problems during file download.
19       */
20      String[] FILTERED_HEADERS = { "cache-control", "expires", "pragma" };
21  
22      /**
23       * Return <code>true</code> if ServletOutputStream has been requested from Table tag.
24       *
25       * @return <code>true</code> if ServletOutputStream has been requested
26       */
27      boolean isOutRequested();
28  
29      /**
30       * If the app server sets the content-type of the response, it is sticky and you will not be able to change it.
31       * Therefore it is intercepted here.
32       *
33       * @return the ContentType that was most recently set
34       */
35      @Override
36      String getContentType();
37  
38      /**
39       * Returns the String representation of the content written to the response.
40       *
41       * @return the content of the response
42       */
43      String getContentAsString();
44  
45  }
