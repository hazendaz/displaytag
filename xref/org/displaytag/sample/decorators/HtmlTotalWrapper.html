25  /**
26   * Same idea implemented in HtmlTableWriter applied to decorators.
27   *
28   * @author Jorge L. Barroso
29   * @version $Revision$ ($Author$)
30   * @see org.displaytag.render.HtmlTableWriter
31   */
32  public class HtmlTotalWrapper extends TotalWrapperTemplate
33  {
34  
35      /**
36       * Write the city in HTML.
37       * @param city City name.
38       * @param total City total.
39       */
40      @Override
41      protected void writeCityTotal(String city, double total)
42      {
43          StringBuffer buffer = this.getStringBuffer();
44          buffer.append("\n<tr>\n<td>&nbsp;</td><td>&nbsp;</td><td><hr/></td>"); //$NON-NLS-1$
45          buffer.append("\n<td>&nbsp;</td></tr>"); //$NON-NLS-1$
46          buffer.append("\n<tr><td>&nbsp;</td>"); //$NON-NLS-1$
47          buffer.append("\n<td align=\"right\"><strong>" //$NON-NLS-1$
48              + city + " Total:</strong></td>\n<td><strong>"); //$NON-NLS-1$
49          buffer.append(total);
50          buffer.append("</strong></td>\n<td>&nbsp;</td>\n</tr>"); //$NON-NLS-1$
51          buffer.append("\n<tr>\n<td colspan=\"4\">&nbsp;\n</td>\n</tr>"); //$NON-NLS-1$
52      }
53  
54      /**
55       * Write the table grand total in HTML.
56       * @param total The table grand total.
57       */
58      @Override
59      protected void writeGrandTotal(double total)
60      {
61          StringBuffer buffer = this.getStringBuffer();
62          buffer.append("<tr><td colspan=\"4\"><hr/></td></tr>"); //$NON-NLS-1$
63          buffer.append("<tr><td>&nbsp;</td>"); //$NON-NLS-1$
64          buffer.append("<td align=\"right\"><strong>Grand Total:</strong></td><td><strong>"); //$NON-NLS-1$
65          buffer.append(total);
66          buffer.append("</strong></td><td>&nbsp;</td></tr>"); //$NON-NLS-1$
67      }
68  }
