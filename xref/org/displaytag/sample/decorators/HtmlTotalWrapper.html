7   package org.displaytag.sample.decorators;
8   
9   /**
10   * Same idea implemented in HtmlTableWriter applied to decorators.
11   *
12   * @see org.displaytag.render.HtmlTableWriter
13   */
14  public class HtmlTotalWrapper extends TotalWrapperTemplate {
15  
16      /**
17       * Write the city in HTML.
18       *
19       * @param city
20       *            City name.
21       * @param total
22       *            City total.
23       */
24      @Override
25      protected void writeCityTotal(String city, double total) {
26          StringBuilder buffer = this.getStringBuilder();
27          buffer.append("\n<tr>\n<td>&nbsp;</td><td>&nbsp;</td><td><hr/></td>"); //$NON-NLS-1$
28          buffer.append("\n<td>&nbsp;</td></tr>"); //$NON-NLS-1$
29          buffer.append("\n<tr><td>&nbsp;</td>"); //$NON-NLS-1$
30          buffer.append("\n<td align=\"right\"><strong>" //$NON-NLS-1$
31                  + city + " Total:</strong></td>\n<td><strong>"); //$NON-NLS-1$
32          buffer.append(total);
33          buffer.append("</strong></td>\n<td>&nbsp;</td>\n</tr>"); //$NON-NLS-1$
34          buffer.append("\n<tr>\n<td colspan=\"4\">&nbsp;\n</td>\n</tr>"); //$NON-NLS-1$
35      }
36  
37      /**
38       * Write the table grand total in HTML.
39       *
40       * @param total
41       *            The table grand total.
42       */
43      @Override
44      protected void writeGrandTotal(double total) {
45          StringBuilder buffer = this.getStringBuilder();
46          buffer.append("<tr><td colspan=\"4\"><hr/></td></tr>"); //$NON-NLS-1$
47          buffer.append("<tr><td>&nbsp;</td>"); //$NON-NLS-1$
48          buffer.append("<td align=\"right\"><strong>Grand Total:</strong></td><td><strong>"); //$NON-NLS-1$
49          buffer.append(total);
50          buffer.append("</strong></td><td>&nbsp;</td></tr>"); //$NON-NLS-1$
51      }
52  }
