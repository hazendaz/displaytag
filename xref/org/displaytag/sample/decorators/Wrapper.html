24  import java.text.DecimalFormat;
25  
26  import org.apache.commons.lang3.time.FastDateFormat;
27  import org.displaytag.decorator.TableDecorator;
28  import org.displaytag.sample.ListObject;
29  
30  
31  /**
32   * This class is a decorator of the TestObjects that we keep in our List. This class provides a number of methods for
33   * formatting data, creating dynamic links, and exercising some aspects of the display:table API functionality.
34   */
35  public class Wrapper extends TableDecorator
36  {
37  
38      /**
39       * FastDateFormat used to format dates in getDate().
40       */
41      private FastDateFormat dateFormat;
42  
43      /**
44       * DecimalFormat used to format money in getMoney().
45       */
46      private DecimalFormat moneyFormat;
47  
48      /**
49       * Creates a new Wrapper decorator who's job is to reformat some of the data located in our TestObject's.
50       */
51      public Wrapper()
52      {
53          super();
54  
55          this.dateFormat = FastDateFormat.getInstance("MM/dd/yy");
56          this.moneyFormat = new DecimalFormat("$ #,###,###.00");
57      }
58  
59      /**
60       * Test method which always returns a null value.
61       *
62       * @return the null value
63       */
64      public String getNullValue()
65      {
66          return null;
67      }
68  
69      /**
70       * Returns the date as a String in MM/dd/yy format.
71       *
72       * @return the date
73       */
74      public String getDate()
75      {
76          return this.dateFormat.format(((ListObject) this.getCurrentRowObject()).getDate());
77      }
78  
79      /**
80       * Returns the money as a String in $ #,###,###.00 format.
81       *
82       * @return the money
83       */
84      public String getMoney()
85      {
86          return this.moneyFormat.format(((ListObject) this.getCurrentRowObject()).getMoney());
87      }
88  
89      /**
90       * Returns the TestObject's ID as a hyperlink that the person can click on and "drill down" for more details.
91       *
92       * @return the link 1
93       */
94      public String getLink1()

97          int index = getListIndex();
98  
99          return "<a href=\"details.jsp?index=" + index + "\">" + object.getId() + "</a>";
100     }
101 
102     /**
103      * Returns an "action bar" of sorts that allow the user to perform various actions on the TestObject based on it's
104      * id.
105      *
106      * @return the link 2
107      */

120             + id

122     }
