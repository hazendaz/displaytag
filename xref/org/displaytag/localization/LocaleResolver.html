7   package org.displaytag.localization;
8   
9   import jakarta.servlet.jsp.PageContext;
10  
11  import java.util.Locale;
12  
13  /**
14   * Handle the selection of the user locale. By default <code>request.getLocale()</code> is used, but specific
15   * implementations can use different way to determine the locale, for example looking at session variables or cookies.
16   */
17  public interface LocaleResolver {
18  
19      /**
20       * Returns the Locale choosen for the given request.
21       *
22       * @param pageContext
23       *            PageContext
24       *
25       * @return a valid Locale (<code>null</code> should never be returned)
26       */
27      Locale resolveLocale(PageContext pageContext);
28  }
