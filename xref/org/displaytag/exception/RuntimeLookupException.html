7   package org.displaytag.exception;
8   
9   import org.displaytag.Messages;
10  import org.displaytag.model.RowSorter;
11  
12  /**
13   * runtime exception thrown during sorting when a checked exception can't be used.
14   */
15  public class RuntimeLookupException extends RuntimeException {
16  
17      /**
18       * Serial ID.
19       */
20      private static final long serialVersionUID = 899149338534L;
21  
22      /**
23       * Instantiates a new runtime lookup exception.
24       *
25       * @param sourceClass
26       *            class where the exception is thrown
27       * @param property
28       *            object property who caused the exception
29       * @param cause
30       *            previous (checked) exception
31       */
32      public RuntimeLookupException(final Class<? extends RowSorter> sourceClass, final String property,
33              final BaseNestableJspTagException cause) {
34          super(Messages.getString("RuntimeLookupException.msg", //$NON-NLS-1$
35                  new Object[] { property, cause.getMessage() }));
36      }
37  
38  }
