7   package org.displaytag.util;

9   import jakarta.servlet.jsp.PageContext;
10  
11  /**
12   * RequestHelperFactory interface.
13   * <p>
14   * Users can specify a custom RequestHelperFactory implementation in <code>displaytag.properties</code>.
15   * <p>
16   * A custom RequestHelperFactory can return a different RequestHelper implementation (the
17   * {@link DefaultRequestHelperFactory}returns instances of {@link DefaultRequestHelper})
18   */
19  public interface RequestHelperFactory {
20  
21      /**
22       * returns a RequestHelper instance for a given request.
23       *
24       * @param pageContext
25       *            PageContext passed by the tag
26       *
27       * @return RequestHelper instance
28       */
29      RequestHelper getRequestHelperInstance(PageContext pageContext);
30  
31  }
