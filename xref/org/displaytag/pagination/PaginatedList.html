7   package org.displaytag.pagination;
8   
9   import java.util.List;
10  
11  import org.displaytag.properties.SortOrderEnum;
12  
13  /**
14   * Interface describing an externally sorted and paginated list.
15   */
16  public interface PaginatedList<T> {
17  
18      /**
19       * Returns the current partial list.
20       *
21       * @return the current partial list
22       */
23      List<T> getList();
24  
25      /**
26       * Returns the page number of the partial list (starts from 1).
27       *
28       * @return the page number of the partial list
29       */
30      int getPageNumber();
31  
32      /**
33       * Returns the number of objects per page. Unless this page is the last one the partial list should thus have a size
34       * equal to the result of this method
35       *
36       * @return the number of objects per page
37       */
38      int getObjectsPerPage();
39  
40      /**
41       * Returns the size of the full list.
42       *
43       * @return the size of the full list
44       */
45      int getFullListSize();
46  
47      /**
48       * Returns the sort criterion used to externally sort the full list.
49       *
50       * @return the sort criterion used to externally sort the full list
51       */
52      String getSortCriterion();
53  
54      /**
55       * Returns the sort direction used to externally sort the full list.
56       *
57       * @return the sort direction used to externally sort the full list
58       */
59      SortOrderEnum getSortDirection();
60  
61      /**
62       * Returns an ID for the search used to get the list. It may be null. Such an ID can be necessary if the full list
63       * is cached, in a way or another (in the session, in the business tier, or anywhere else), to be able to retrieve
64       * the full list from the cache
65       *
66       * @return the search ID
67       */
68      String getSearchId();
69  }
