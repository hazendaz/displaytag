7   package org.displaytag.test;
8   
9   import java.util.Comparator;
10  
11  import org.apache.commons.beanutils2.ConvertUtils;
12  
13  /**
14   * Sorts 2 numbers, converted from objects using beanutils Converters.
15   */
16  public class NumberComparator implements Comparator<Object> {
17  
18      /**
19       * Compare.
20       *
21       * @param obj1
22       *            the obj 1
23       * @param obj2
24       *            the obj 2
25       *
26       * @return the int
27       *
28       * @see Comparator#compare(Object, Object)
29       */
30      @Override
31      public int compare(final Object obj1, final Object obj2) {
32          double dbl1 = 0;
33          if (obj1 instanceof Number) {
34              dbl1 = ((Number) obj1).doubleValue();
35          } else if (obj1 != null) {
36              dbl1 = ((Number) ConvertUtils.convert(obj1.toString(), Number.class)).doubleValue();
37          }
38  
39          double dbl2 = 0;
40          if (obj2 instanceof Number) {
41              dbl2 = ((Number) obj2).doubleValue();
42          } else if (obj1 != null) {
43              dbl2 = ((Number) ConvertUtils.convert(obj2.toString(), Number.class)).doubleValue();
44          }
45  
46          return Double.valueOf(dbl1).compareTo(Double.valueOf(dbl2));
47      }
48  }
