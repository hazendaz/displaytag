7   package org.displaytag.util;
8   
9   import jakarta.servlet.http.HttpServletRequest;
10  import jakarta.servlet.http.HttpServletResponse;
11  import jakarta.servlet.jsp.PageContext;
12  
13  /**
14   * Default RequestHelperFactory implementation. Returns instaces of {@link DefaultRequestHelper}.
15   */
16  public class DefaultRequestHelperFactory implements RequestHelperFactory {
17  
18      /**
19       * Gets the request helper instance.
20       *
21       * @param pageContext
22       *            the page context
23       *
24       * @return the request helper instance
25       *
26       * @see org.displaytag.util.RequestHelperFactory#getRequestHelperInstance(jakarta.servlet.jsp.PageContext)
27       */
28      @Override
29      public RequestHelper getRequestHelperInstance(final PageContext pageContext) {
30          return new DefaultRequestHelper((HttpServletRequest) pageContext.getRequest(),
31                  (HttpServletResponse) pageContext.getResponse());
32      }
33  
34  }
