48  /**
49   * A table writer that formats table as and writes it to an iText document.
50   * @author Jorge L. Barroso
51   * @version $Id$
52   * @see org.displaytag.render.TableWriterTemplate
53   */
54  public class ItextTableWriter extends TableWriterAdapter
55  {
56  
57      /**
58       * iText representation of the table.
59       */
60      private PdfPTable table;
61  
62      /**
63       * iText document to which the table is written.
64       */
65      private Document document;
66  
67      /**
68       * The default font used in the document.
69       */
70      private Font defaultFont;
71  
72      /**
73       * This table writer uses an iText table and document to do its work.
74       * @param table iText representation of the table.
75       * @param document iText document to which the table is written.
76       */
77      public ItextTableWriter(PdfPTable table, Document document)
78      {
79          this.table = table;
80          this.document = document;
81      }
82  
83      /**
84       * Initialize the main info holder table, like the appropriate number of columns.
85       * @param model The table being represented as iText.
86       * @see org.displaytag.render.TableWriterTemplate#writeTableOpener(org.displaytag.model.TableModel)
87       */
88      @Override
89      protected void writeTableOpener(TableModel model)
90      {
91          this.table.getDefaultCell().setVerticalAlignment(Element.ALIGN_TOP);
92          this.table.setWidthPercentage(100);
93          this.defaultFont = this.getTableFont();
94      }
95  
96      /**
97       * Obtain the font used to render text in the table; Meant to be overriden if a different font is desired.
98       * @return The font used to render text in the table.
99       */
100     protected Font getTableFont()
101     {
102         return FontFactory.getFont(FontFactory.HELVETICA, 10, Font.NORMAL, new BaseColor(0x00, 0x00, 0x00));
103     }
104 
105     /**
106      * Write the table's caption to a iText document.
107      * @see org.displaytag.render.TableWriterTemplate#writeCaption(org.displaytag.model.TableModel)
108      */
109     @Override
110     protected void writeCaption(TableModel model) throws Exception
111     {
112         this.decorateCaption(model);
113     }
114 
115     /**
116      * Writes the table caption according to a set style.
117      * @param model The table model containing the caption.
118      * @throws DocumentException If an error occurrs while decorating the caption.
119      */
120     private void decorateCaption(TableModel model) throws DocumentException
121     {
122         Paragraph caption = new Paragraph(new Chunk(model.getCaption(), this.getCaptionFont()));
123         caption.setAlignment(this.getCaptionHorizontalAlignment());
124         this.document.add(caption);
125     }
126 
127     /**
128      * Obtain the caption font; Meant to be overriden if a different style is desired.
129      * @return The caption font.
130      */
131     protected Font getCaptionFont()
132     {
133         return FontFactory.getFont(FontFactory.HELVETICA, 17, Font.BOLD, new BaseColor(0x00, 0x00, 0x00));
134     }

136     /**
137      * Obtain the caption horizontal alignment; Meant to be overriden if a different style is desired.
138      * @return The caption horizontal alignment.
139      */
140     protected int getCaptionHorizontalAlignment()

142         return Element.ALIGN_CENTER;
143     }

145     /**
146      * Write the table's header columns to an iText document.
147      * @see org.displaytag.render.TableWriterTemplate#writeTableHeader(org.displaytag.model.TableModel)
148      * @throws DocumentException if an error occurs while writing header.
149      */
150     @Override
151     protected void writeTableHeader(TableModel model) throws DocumentException
152     {
153         Iterator<HeaderCell> iterator = model.getHeaderCellList().iterator();
154 
155         float[] widths = new float[model.getNumberOfColumns()];
156         for (int i = 0; iterator.hasNext(); i++)
157         {
158             HeaderCell headerCell = iterator.next();
159             widths[i] = this.getCellWidth(headerCell);
160 
161             String columnHeader = headerCell.getTitle();
162 
163             if (columnHeader == null)
164             {
165                 columnHeader = StringUtils.capitalize(headerCell.getBeanPropertyName());
166             }
167 
168             PdfPCell hdrCell = this.getHeaderCell(columnHeader);
169             this.table.addCell(hdrCell);
170         }
171         this.table.setHeaderRows(1);
172         this.table.setWidths(widths);
173     }
174 
175     /**
176      * Returns the maximum size of all values in this column.
177      * @param headerCell Header cell for this column.
178      * @return The maximum size of all values in this column.
179      */
180     private float getCellWidth(HeaderCell headerCell)
181     {
182         int maxWidth = headerCell.getMaxLength();
183         return (maxWidth > 0) ? maxWidth : headerCell.getTitle().length();
184     }
185 
186     /**
187      * @see org.displaytag.render.TableWriterTemplate#writePostBodyFooter(org.displaytag.model.TableModel)
188      * @throws DocumentException if an error occurs while writing post-body footer.
189      */
190     @Override
191     protected void writePostBodyFooter(TableModel model) throws DocumentException
192     {
193         Chunk cellContent = new Chunk(model.getFooter(), this.getFooterFont());
194         this.setFooterFontStyle(cellContent);
195         PdfPCell cell = new PdfPCell(new Phrase(cellContent));
196         cell.setLeading(8, 0);
197         cell.setBackgroundColor(this.getFooterBackgroundColor());
198         cell.setHorizontalAlignment(this.getFooterHorizontalAlignment());
199         cell.setColspan(model.getNumberOfColumns());
200         this.table.addCell(cell);
201     }
202

204      * Obtain the footer background color; Meant to be overriden if a different style is desired.
205      * @return The footer background color.
206      */
207     protected BaseColor getFooterBackgroundColor()
208     {
209         return new BaseColor(0xce, 0xcf, 0xce);
210     }
211 
212     /**
213      * Obtain the footer horizontal alignment; Meant to be overriden if a different style is desired.

216     protected int getFooterHorizontalAlignment()

218         return Element.ALIGN_LEFT;
219     }
220

222      * Set the font style used to render the header text; Meant to be overridden if a different header style is desired.
223      * @param cellContent The header content whose font will be modified.

227         this.setBoldStyle(cellContent, this.getFooterFontColor());

237     }

242      */
243     protected Font getFooterFont()

245         return FontFactory.getFont(FontFactory.HELVETICA, 10);
246     }

248     /**
249      * Decorators that help render the table to an iText document must implement ItextDecorator.
250      * @see org.displaytag.render.TableWriterTemplate#writeDecoratedRowStart(org.displaytag.model.TableModel)
251      */
252     @Override
253     protected void writeDecoratedRowStart(TableModel model)
254     {
255         TableDecorator decorator = model.getTableDecorator();
256         if (decorator instanceof ItextDecorator)
257         {
258             ItextDecorator idecorator = (ItextDecorator) decorator;
259             idecorator.setTable(this.table);
260             idecorator.setFont(this.defaultFont);
261         }
262         decorator.startRow();
263     }
264 
265     /**
266      * @see org.displaytag.render.TableWriterTemplate#writeDecoratedRowFinish(org.displaytag.model.TableModel)
267      */
268     @Override
269     protected void writeDecoratedRowFinish(TableModel model) throws Exception
270     {
271         model.getTableDecorator().finishRow();
272     }

285      * Write a column's value to a iText document.

289     protected void writeColumnValue(Object value, Column column) throws BadElementException

292     }

295      * @see org.displaytag.render.TableWriterTemplate#writeDecoratedTableFinish(org.displaytag.model.TableModel)

297     @Override

299     {
300         model.getTableDecorator().finish();
301     }

303     /**
304      * Returns a formatted cell for the given value.
305      * @param value cell value
306      * @return Cell
307      * @throws BadElementException if errors occurs while generating content.
308      */
309     private PdfPCell getCell(Object value)
310     {
311         PdfPCell cell = new PdfPCell(new Phrase(new Chunk(StringUtils.trimToEmpty(value != null
312             ? value.toString()
313             : StringUtils.EMPTY), this.defaultFont)));
314         cell.setVerticalAlignment(Element.ALIGN_TOP);
315         cell.setLeading(8, 0);
316         return cell;
317     }
318 
319     /**
320      * Obtain a header cell.
321      * @param value Cell content.
322      * @return A header cell with the given content.
323      */
324     private PdfPCell getHeaderCell(String value)
325     {
326         Chunk cellContent = new Chunk(value, this.getHeaderFont());
327         setHeaderFontStyle(cellContent);
328         PdfPCell cell = new PdfPCell(new Phrase(cellContent));
329         cell.setLeading(8, 0);
330         cell.setHorizontalAlignment(this.getHeaderHorizontalAlignment());
331         cell.setBackgroundColor(this.getHeaderBackgroundColor());
332         return cell;
333     }

382     /**
383      * Makes chunk content bold.
384      * @param chunk The chunk whose content is to be rendered bold.
385      * @param color The font color desired.
386      */
387     private void setBoldStyle(Chunk chunk, BaseColor color)
388     {
389         Font font = chunk.getFont();
390         chunk.setFont(FontFactory.getFont(font.getFamilyname(), font.getSize(), Font.BOLD, color));
391     }
