7   package org.displaytag.localization;
8   
9   import jakarta.servlet.jsp.PageContext;
10  import jakarta.servlet.jsp.tagext.Tag;
11  
12  /**
13   * Interface for resource providers. Given a <code>resourceKey</code> and a <code>defaultValue</code>, a resource
14   * provider returns the String which should be displayed. For column headers <code>resourceKey</code> will be the value
15   * of the <code>titleKey</code> attribute, <code>defaultValue</code> will be the value of the <code>property</code>
16   * attribute. Different resource providers can be plugged using the displaytag.properties file.
17   */
18  public interface I18nResourceProvider {
19  
20      /**
21       * Returns a localized String. A resource provider is free to use both <code>resourceKey</code> or
22       * <code>defaultValue</code> for the lookup. For example in column titles <code>defaultValue</code> is the value of
23       * the <code>property</code> attribute and can be used as a default if <code>titleKey</code> is not specified.
24       *
25       * @param resourceKey
26       *            used-specified resource key
27       * @param defaultValue
28       *            default or fallback value
29       * @param tag
30       *            calling tag (TableTag), which can be used to find needed ancestor tags
31       * @param context
32       *            jsp page context
33       *
34       * @return localized String.
35       */
36      String getResource(String resourceKey, String defaultValue, Tag tag, PageContext context);
37  
38  }
