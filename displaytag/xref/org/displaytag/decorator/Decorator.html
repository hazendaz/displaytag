37  /**
38   * <p>
39   * This class provides some basic functionality for all objects which serve as decorators for the objects in the List
40   * being displayed.
41   * <p>
42   * <p>
43   * Decorator should never be subclassed directly. Use TableDecorator instead
44   * </p>
45   * @author mraible
46   * @author Fabrizio Giustina
47   * @version $Revision$ ($Author$)
48   */
49  abstract class Decorator
50  {
51  
52      /**
53       * Char used to separate class name and property in the cache key.
54       */
55      private static final char CLASS_PROPERTY_SEPARATOR = '#';
56  
57      /**
58       * property info cache contains classname#propertyname Strings as keys and Booleans as values.
59       */
60      private static Map<String, Boolean> propertyMap = new ConcurrentHashMap<String, Boolean>();
61  
62      /**
63       * page context.
64       */
65      private PageContext pageContext;
66  
67      /**
68       * decorated object. Usually a List
69       */
70      private Object decoratedObject;
71  
72      /**
73       * The table model.
74       * @since 1.1
75       */
76      protected TableModel tableModel;
77  
78      /**
79       * Initialize the TableTecorator instance.
80       * @param pageContext PageContext
81       * @param decorated decorated object (usually a list)
82       * @deprecated use #init(PageContext, Object, TableModel)
83       * @see #init(PageContext, Object, TableModel)
84       */
85      @Deprecated
86      public void init(PageContext pageContext, Object decorated)
87      {
88          this.pageContext = pageContext;
89          this.decoratedObject = decorated;
90      }
91  
92      /**
93       * Initialize the TableTecorator instance.
94       * @param pageContext PageContext
95       * @param decorated decorated object (usually a list)
96       * @param tableModel table model
97       */
98      public void init(PageContext pageContext, Object decorated, TableModel tableModel)
99      {
100         // temporary used for backward (source) compatibility
101         init(pageContext, decorated);
102         this.tableModel = tableModel;
103     }
104 
105     /**
106      * returns the page context.
107      * @return PageContext
108      */
109     public PageContext getPageContext()
110     {
111         return this.pageContext;
112     }
113 
114     /**
115      * returns the decorated object.
116      * @return Object
117      */
118     public Object getDecoratedObject()
119     {
120         return this.decoratedObject;
121     }
122 
123     /**
124      * Called at the end of evaluation to clean up instance variable. A subclass of Decorator can override this method
125      * but should always call super.finish() before return
126      */
127     public void finish()
128     {
129         this.pageContext = null;
130         this.decoratedObject = null;
131     }
132 
133     /**
134      * Check if a getter exists for a given property. Uses cached info if property has already been requested. This
135      * method only check for a simple property, if pPropertyName contains multiple tokens only the first part is
136      * evaluated
137      * @param propertyName name of the property to check
138      * @return boolean true if the decorator has a getter for the given property
139      */
140     public boolean hasGetterFor(String propertyName)
141     {
142         String simpleProperty = propertyName;
143 
144         // get the simple (not nested) bean property
145         int indexOfDot = simpleProperty.indexOf('.');
146         if (indexOfDot > 0)
147         {
148             simpleProperty = simpleProperty.substring(0, indexOfDot);
149         }
150 
151         Boolean cachedResult = propertyMap.get(getClass().getName() + CLASS_PROPERTY_SEPARATOR + simpleProperty);
152 
153         if (cachedResult != null)
154         {
155             return cachedResult.booleanValue();
156         }
157 
158         // not already cached... check
159         boolean hasGetter = searchGetterFor(propertyName);
160 
161         // save in cache
162         propertyMap.put(getClass().getName() + CLASS_PROPERTY_SEPARATOR + simpleProperty, hasGetter);
163 
164         // and return
165         return hasGetter;
166 
167     }
168 
169     /**
170      * Looks for a getter for the given property using introspection.
171      * @param propertyName name of the property to check
172      * @return boolean true if the decorator has a getter for the given property
173      */
174     public boolean searchGetterFor(String propertyName)
175     {
176 
177         boolean result = false;
178 
179         try
180         {
181             // using getPropertyType instead of isReadable since isReadable doesn't support mapped properties.
182             // Note that this method usually returns null if a property is not found and doesn't throw any exception
183             // also for non existent properties
184             PropertyDescriptor pd = PropertyUtils.getPropertyDescriptor(this, propertyName);
185 
186             if (pd != null)
187             {
188                 // double check, see tests in TableDecoratorTest
189                 if (pd instanceof MappedPropertyDescriptor)
190                 {
191                     result = ((MappedPropertyDescriptor) pd).getMappedReadMethod() != null;
192                 }
193                 else if (pd instanceof IndexedPropertyDescriptor)
194                 {
195                     result = ((IndexedPropertyDescriptor) pd).getIndexedReadMethod() != null;
196                 }
197                 else
198                 {
199                     result = pd.getReadMethod() != null;
200                 }
201             }
202         }
203         catch (IllegalAccessException e)
204         {
205             // ignore
206         }
207         catch (InvocationTargetException e)
208         {
209             // ignore
210         }
211         catch (NoSuchMethodException e)
212         {
213             // ignore
214         }
215 
216         return result;
217 
218     }
