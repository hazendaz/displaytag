118     /**
119      * We cache the styles; they are expensive to construct.
120      * @param properties props for this run
121      */
122     public void initCellStyles(TableProperties properties)
123     {
124         // Integer
125         HSSFCellStyle style = getNewCellStyle();
126         style.setAlignment(CellStyle.ALIGN_RIGHT);
127         style.setDataFormat(HSSFDataFormat.getBuiltinFormat(properties.getProperty(ExcelUtils.EXCEL_FORMAT_INTEGER)));
128         this.cellStyles.put(STYLE_INTEGER, style);
129 
130         // NUMBER
131         style = getNewCellStyle();
132         style.setAlignment(CellStyle.ALIGN_RIGHT);
133         style.setDataFormat(HSSFDataFormat.getBuiltinFormat(properties.getProperty(ExcelUtils.EXCEL_FORMAT_NUMBER)));
134         this.cellStyles.put(STYLE_NUMBER, style);
135 
136         // style = HSSFDataFormat.getBuiltinFormat("0.00%");
137 
138         // Date
139         style = getNewCellStyle();
140         style.setAlignment(CellStyle.ALIGN_RIGHT);
141         style.setDataFormat(HSSFDataFormat.getBuiltinFormat(properties.getProperty(ExcelUtils.EXCEL_FORMAT_DATE)));
142         style.setAlignment(CellStyle.ALIGN_RIGHT);
143         this.cellStyles.put(STYLE_DATE, style);
144 
145         // Long text
146         style = getNewCellStyle(); // http://jakarta.apache.org/poi/hssf/quick-guide.html#NewLinesInCells
147         style.setWrapText(true);
148         this.cellStyles.put(STYLE_LONGSTRING, style);
149 
150         // Regular text
151         this.cellStyles.put(STYLE_STRING, getNewCellStyle());
152 
153         this.wrapAt = Integer.valueOf(properties.getProperty(ExcelUtils.EXCEL_WRAPAT));
154     }

222     // patch from Karsten Voges
223     /**
224      * Escape certain values that are not permitted in excel cells.
225      * @param rawValue the object value
226      * @return the escaped value
227      */
228     public static String escapeColumnValue(Object rawValue)
229     {
230         if (rawValue == null)
231         {
232             return null;
233         }
234         // str = Patterns.replaceAll(str, "(\\r\\n|\\r|\\n|\\n\\r)\\s*", "");
235         String returnString = rawValue.toString();
236         // escape the String to get the tabs, returns, newline explicit as \t \r \n
237         returnString = StringEscapeUtils.escapeJava(StringUtils.trimToEmpty(returnString));
238         // remove tabs, insert four whitespaces instead
239         returnString = StringUtils.replace(StringUtils.trim(returnString), "\\t", "    ");
240         // remove the return, only newline valid in excel
241         returnString = StringUtils.replace(StringUtils.trim(returnString), "\\r", " ");
242         // unescape so that \n gets back to newline
243         returnString = StringEscapeUtils.unescapeJava(returnString);
244         return returnString;
245     }
