7   package org.displaytag.export;
8   
9   import org.displaytag.model.TableModel;
10  
11  /**
12   * Interface for export classes. ExportViewFactory is responsible for registering and initialization of export views. A
13   * default, no parameters constructor is required. The <code>setParameters()</code> method is guarantee to be called
14   * before any other operation.
15   */
16  public interface ExportView {
17  
18      /**
19       * initialize the parameters needed for export. The method is guarantee be called before <code>doExport()</code> and
20       * <code>getMimeType()</code>. Classes implementing ExportView should reset any instance field previously set when
21       * this method is called, in order to support instance reusing.
22       *
23       * @param tableModel
24       *            TableModel to render
25       * @param exportFullList
26       *            boolean export full list?
27       * @param includeHeader
28       *            should header be included in export?
29       * @param decorateValues
30       *            should output be decorated?
31       */
32      void setParameters(TableModel tableModel, boolean exportFullList, boolean includeHeader, boolean decorateValues);
33  
34      /**
35       * MimeType to return.
36       *
37       * @return String mime type
38       */
39      String getMimeType();
40  
41  }
