7   package org.displaytag.exception;
8   
9   /**
10   * Basic wrapper for checked exceptions.
11   */
12  public class WrappedRuntimeException extends BaseNestableRuntimeException {
13  
14      /**
15       * Serial ID.
16       */
17      private static final long serialVersionUID = 899149338534L;
18  
19      /**
20       * Instantiate a new WrappedRuntimeException.
21       *
22       * @param source
23       *            Class where the exception is generated
24       * @param cause
25       *            Original exception
26       */
27      public WrappedRuntimeException(final Class<?> source, final Throwable cause) {
28          super(source, cause.getMessage(), cause);
29      }
30  
31      /**
32       * Gets the severity.
33       *
34       * @return the severity
35       *
36       * @see org.displaytag.exception.BaseNestableRuntimeException#getSeverity()
37       */
38      @Override
39      public SeverityEnum getSeverity() {
40          return SeverityEnum.WARN;
41      }
42  
43  }
