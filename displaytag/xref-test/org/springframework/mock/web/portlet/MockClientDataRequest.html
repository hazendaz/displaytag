17  package org.springframework.mock.web.portlet;
18  
19  import java.io.BufferedReader;
20  import java.io.ByteArrayInputStream;
21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.io.InputStreamReader;
24  import java.io.Reader;
25  import java.io.UnsupportedEncodingException;
26  import java.nio.charset.Charset;
27  import java.nio.charset.StandardCharsets;
28  import java.util.Collection;
29  
30  import javax.portlet.ClientDataRequest;
31  import javax.portlet.PortalContext;
32  import javax.portlet.PortletContext;
33  import javax.portlet.PortletException;
34  import javax.servlet.http.Part;
35  
36  /**
37   * Mock implementation of the {@link javax.portlet.ClientDataRequest} interface.
38   *
39   * @author Juergen Hoeller
40   * @since 3.0
41   */
42  public class MockClientDataRequest extends MockPortletRequest implements ClientDataRequest {
43  
44  	/** The character encoding. */
45  	private String characterEncoding;
46  
47  	/** The content. */
48  	private byte[] content;
49  
50  	/** The content type. */
51  	private String contentType;
52  
53  	/** The method. */
54  	private String method;
55  
56  
57  	/**
58  	 * Create a new MockClientDataRequest with a default {@link MockPortalContext}
59  	 * and a default {@link MockPortletContext}.
60  	 * @see org.springframework.mock.web.portlet.MockPortalContext
61  	 * @see org.springframework.mock.web.portlet.MockPortletContext
62  	 */
63  	public MockClientDataRequest() {
64  		super();
65  	}
66  
67  	/**
68  	 * Create a new MockClientDataRequest with a default {@link MockPortalContext}.
69  	 * @param portletContext the PortletContext that the request runs in
70  	 */
71  	public MockClientDataRequest(PortletContext portletContext) {
72  		super(portletContext);
73  	}
74  
75  	/**
76  	 * Create a new MockClientDataRequest.
77  	 * @param portalContext the PortalContext that the request runs in
78  	 * @param portletContext the PortletContext that the request runs in
79  	 */
80  	public MockClientDataRequest(PortalContext portalContext, PortletContext portletContext) {
81  		super(portalContext, portletContext);
82  	}
83  
84  
85  	/**
86  	 * Sets the content.
87  	 *
88  	 * @param content the new content
89  	 */
90  	public void setContent(byte[] content) {
91  		this.content = content;
92  	}
93  
94  	@Override
95  	public InputStream getPortletInputStream() throws IOException {
96  		if (this.content != null) {
97  			return new ByteArrayInputStream(this.content);
98  		}
99  		else {
100 			return null;
101 		}
102 	}
103 
104 	@Override
105 	public void setCharacterEncoding(String characterEncoding) {
106 		this.characterEncoding = characterEncoding;
107 	}
108 
109 	@Override
110 	public BufferedReader getReader() throws UnsupportedEncodingException {
111 		if (this.content != null) {
112 			InputStream sourceStream = new ByteArrayInputStream(this.content);
113 			Reader sourceReader = (this.characterEncoding != null) ?
114 				new InputStreamReader(sourceStream, Charset.forName(this.characterEncoding)) : new InputStreamReader(sourceStream, StandardCharsets.UTF_8);
115 			return new BufferedReader(sourceReader);
116 		}
117 		else {
118 			return null;
119 		}
120 	}
121 
122 	@Override
123 	public String getCharacterEncoding() {
124 		return this.characterEncoding;
125 	}
126 
127 	/**
128 	 * Sets the content type.
129 	 *
130 	 * @param contentType the new content type
131 	 */
132 	public void setContentType(String contentType) {
133 		this.contentType = contentType;
134 	}
135 
136 	@Override
137 	public String getContentType() {
138 		return this.contentType;
139 	}
140 
141 	@Override
142 	public int getContentLength() {
143 		return (this.content != null ? content.length : -1);
144 	}
145 
146 	/**
147 	 * Sets the method.
148 	 *
149 	 * @param method the new method
150 	 */
151 	public void setMethod(String method) {
152 		this.method = method;
153 	}
154 
155 	@Override
156 	public String getMethod() {
157 		return this.method;
158 	}
159 
160   @Override
161   public long getContentLengthLong() {
162     return content.length;
163   }
164 
165   @Override
166   public Part getPart(String name) throws IOException, PortletException {
167     // TODO Auto-generated method stub
168     return null;
169   }
170 
171   @Override
172   public Collection<Part> getParts() throws IOException, PortletException {
173     // TODO Auto-generated method stub
174     return null;
175   }
176 
177 }
