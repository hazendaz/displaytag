17  package org.springframework.mock.web.portlet;
18  
19  import java.io.IOException;
20  import javax.portlet.PortletException;
21  import javax.portlet.PortletRequest;
22  import javax.portlet.PortletRequestDispatcher;
23  import javax.portlet.PortletResponse;
24  import javax.portlet.RenderRequest;
25  import javax.portlet.RenderResponse;
26  
27  import org.junit.jupiter.api.Assertions;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  /**
32   * Mock implementation of the {@link javax.portlet.PortletRequestDispatcher} interface.
33   *
34   * @author John A. Lewis
35   * @author Juergen Hoeller
36   * @since 2.0
37   */
38  public class MockPortletRequestDispatcher implements PortletRequestDispatcher {
39  
40  	/** The logger. */
41  	private final Logger logger = LoggerFactory.getLogger(getClass());
42  
43  	/** The url. */
44  	private final String url;
45  
46  
47  	/**
48  	 * Create a new MockPortletRequestDispatcher for the given URL.
49  	 * @param url the URL to dispatch to.
50  	 */
51  	public MockPortletRequestDispatcher(String url) {
52  		Assertions.assertNotNull(url, "URL must not be null");
53  		this.url = url;
54  	}
55  
56  
57  	@Override
58  	public void include(RenderRequest request, RenderResponse response) throws PortletException, IOException {
59  		include((PortletRequest) request, (PortletResponse) response);
60  	}
61  
62  	@Override
63  	public void include(PortletRequest request, PortletResponse response) throws PortletException, IOException {
64  		Assertions.assertNotNull(request, "Request must not be null");
65  		Assertions.assertNotNull(response, "Response must not be null");
66  		if (!(response instanceof MockMimeResponse)) {
67  			throw new IllegalArgumentException("MockPortletRequestDispatcher requires MockMimeResponse");
68  		}
69  		((MockMimeResponse) response).setIncludedUrl(this.url);
70  		if (logger.isDebugEnabled()) {
71  			logger.debug("MockPortletRequestDispatcher: including URL [" + this.url + "]");
72  		}
73  	}
74  
75  	@Override
76  	public void forward(PortletRequest request, PortletResponse response) throws PortletException, IOException {
77  		Assertions.assertNotNull(request, "Request must not be null");
78  		Assertions.assertNotNull(response, "Response must not be null");
79  		if (!(response instanceof MockMimeResponse)) {
80  			throw new IllegalArgumentException("MockPortletRequestDispatcher requires MockMimeResponse");
81  		}
82  		((MockMimeResponse) response).setForwardedUrl(this.url);
83  		if (logger.isDebugEnabled()) {
84  			logger.debug("MockPortletRequestDispatcher: forwarding to URL [" + this.url + "]");
85  		}
86  	}
87  
88  }
