37  /**
38   * Mock implementation of the {@link javax.portlet.PortletConfig} interface.
39   *
40   * @author John A. Lewis
41   * @author Juergen Hoeller
42   * @since 2.0
43   */
44  public class MockPortletConfig implements PortletConfig {
45  
46  	/** The portlet context. */
47  	private final PortletContext portletContext;
48  
49  	/** The portlet name. */
50  	private final String portletName;
51  
52  	/** The resource bundles. */
53  	private final Map<Locale, ResourceBundle> resourceBundles = new HashMap<Locale, ResourceBundle>();
54  
55  	/** The init parameters. */
56  	private final Map<String, String> initParameters = new LinkedHashMap<String, String>();
57  
58  	/** The public render parameter names. */
59  	private final Set<String> publicRenderParameterNames = new LinkedHashSet<String>();
60  
61  	/** The default namespace. */
62  	private String defaultNamespace = XMLConstants.NULL_NS_URI;
63  
64  	/** The publishing event Q names. */
65  	private final Set<QName> publishingEventQNames = new LinkedHashSet<QName>();
66  
67  	/** The processing event Q names. */
68  	private final Set<QName> processingEventQNames = new LinkedHashSet<QName>();
69  
70  	/** The supported locales. */
71  	private final Set<Locale> supportedLocales = new LinkedHashSet<Locale>();
72  
73  	/** The container runtime options. */
74  	private final Map<String, String[]> containerRuntimeOptions = new LinkedHashMap<String, String[]>();
75  
76  
77  	/**
78  	 * Create a new MockPortletConfig with a default {@link MockPortletContext}.
79  	 */
80  	public MockPortletConfig() {
81  		this(null, "");
82  	}
83  
84  	/**
85  	 * Create a new MockPortletConfig with a default {@link MockPortletContext}.
86  	 * @param portletName the name of the portlet
87  	 */
88  	public MockPortletConfig(String portletName) {
89  		this(null, portletName);
90  	}
91  
92  	/**
93  	 * Create a new MockPortletConfig.
94  	 * @param portletContext the PortletContext that the portlet runs in
95  	 */
96  	public MockPortletConfig(PortletContext portletContext) {
97  		this(portletContext, "");
98  	}
99  
100 	/**
101 	 * Create a new MockPortletConfig.
102 	 * @param portletContext the PortletContext that the portlet runs in
103 	 * @param portletName the name of the portlet
104 	 */
105 	public MockPortletConfig(PortletContext portletContext, String portletName) {
106 		this.portletContext = (portletContext != null ? portletContext : new MockPortletContext());
107 		this.portletName = portletName;
108 	}
109 
110 
111 	@Override
112 	public String getPortletName() {
113 		return this.portletName;
114 	}
115 
116 	@Override
117 	public PortletContext getPortletContext() {
118 		return this.portletContext;
119 	}
120 
121 	/**
122 	 * Sets the resource bundle.
123 	 *
124 	 * @param locale the locale
125 	 * @param resourceBundle the resource bundle
126 	 */
127 	public void setResourceBundle(Locale locale, ResourceBundle resourceBundle) {
128 		Assertions.assertNotNull(locale, "Locale must not be null");
129 		this.resourceBundles.put(locale, resourceBundle);
130 	}
131 
132 	@Override
133 	public ResourceBundle getResourceBundle(Locale locale) {
134 		Assertions.assertNotNull(locale, "Locale must not be null");
135 		return this.resourceBundles.get(locale);
136 	}
137 
138 	/**
139 	 * Adds the init parameter.
140 	 *
141 	 * @param name the name
142 	 * @param value the value
143 	 */
144 	public void addInitParameter(String name, String value) {
145 		Assertions.assertNotNull(name, "Parameter name must not be null");
146 		this.initParameters.put(name, value);
147 	}
148 
149 	@Override
150 	public String getInitParameter(String name) {
151 		Assertions.assertNotNull(name, "Parameter name must not be null");
152 		return this.initParameters.get(name);
153 	}
154 
155 	@Override
156 	public Enumeration<String> getInitParameterNames() {
157 		return Collections.enumeration(this.initParameters.keySet());
158 	}
159 
160 	/**
161 	 * Adds the public render parameter name.
162 	 *
163 	 * @param name the name
164 	 */
165 	public void addPublicRenderParameterName(String name) {
166 		this.publicRenderParameterNames.add(name);
167 	}
168 
169 	@Override
170 	public Enumeration<String> getPublicRenderParameterNames() {
171 		return Collections.enumeration(this.publicRenderParameterNames);
172 	}
173 
174 	/**
175 	 * Sets the default namespace.
176 	 *
177 	 * @param defaultNamespace the new default namespace
178 	 */
179 	public void setDefaultNamespace(String defaultNamespace) {
180 		this.defaultNamespace = defaultNamespace;
181 	}
182 
183 	@Override
184 	public String getDefaultNamespace() {
185 		return this.defaultNamespace;
186 	}
187 
188 	/**
189 	 * Adds the publishing event Q name.
190 	 *
191 	 * @param name the name
192 	 */
193 	public void addPublishingEventQName(QName name) {
194 		this.publishingEventQNames.add(name);
195 	}
196 
197 	@Override
198 	public Enumeration<QName> getPublishingEventQNames() {
199 		return Collections.enumeration(this.publishingEventQNames);
200 	}
201 
202 	/**
203 	 * Adds the processing event Q name.
204 	 *
205 	 * @param name the name
206 	 */
207 	public void addProcessingEventQName(QName name) {
208 		this.processingEventQNames.add(name);
209 	}
210 
211 	@Override
212 	public Enumeration<QName> getProcessingEventQNames() {
213 		return Collections.enumeration(this.processingEventQNames);
214 	}
215 
216 	/**
217 	 * Adds the supported locale.
218 	 *
219 	 * @param locale the locale
220 	 */
221 	public void addSupportedLocale(Locale locale) {
222 		this.supportedLocales.add(locale);
223 	}
224 
225 	@Override
226 	public Enumeration<Locale> getSupportedLocales() {
227 		return Collections.enumeration(this.supportedLocales);
228 	}
229 
230 	/**
231 	 * Adds the container runtime option.
232 	 *
233 	 * @param key the key
234 	 * @param value the value
235 	 */
236 	public void addContainerRuntimeOption(String key, String value) {
237 		this.containerRuntimeOptions.put(key, new String[] {value});
238 	}
239 
240 	/**
241 	 * Adds the container runtime option.
242 	 *
243 	 * @param key the key
244 	 * @param values the values
245 	 */
246 	public void addContainerRuntimeOption(String key, String[] values) {
247 		this.containerRuntimeOptions.put(key, values);
248 	}
249 
250 	@Override
251 	public Map<String, String[]> getContainerRuntimeOptions() {
252 		return Collections.unmodifiableMap(this.containerRuntimeOptions);
253 	}
