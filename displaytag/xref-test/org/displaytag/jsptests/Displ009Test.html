7   package org.displaytag.jsptests;
8   
9   import com.meterware.httpunit.GetMethodWebRequest;
10  import com.meterware.httpunit.WebRequest;
11  import com.meterware.httpunit.WebResponse;
12  
13  import org.displaytag.test.DisplaytagCase;
14  import org.junit.jupiter.api.Assertions;
15  import org.junit.jupiter.api.Test;
16  
17  /**
18   * Tests for DISPL-9 - Send user back to Page 1 on Desc/Asc.
19   */
20  class Displ009Test extends DisplaytagCase {
21  
22      /**
23       * Gets the jsp name.
24       *
25       * @return the jsp name
26       *
27       * @see org.displaytag.test.DisplaytagCase#getJspName()
28       */
29      @Override
30      public String getJspName() {
31          return "DISPL-009.jsp";
32      }
33  
34      /**
35       * Verifies that the generated page contains a table with the expected number of columns.
36       *
37       * @throws Exception
38       *             any exception thrown during test.
39       */
40      @Override
41      @Test
42      public void doTest() throws Exception {
43          final WebRequest request = new GetMethodWebRequest(this.getJspUrl(this.getJspName()));
44          WebResponse response;
45  
46          // step 0
47          response = this.runner.getResponse(request);
48  
49          // link[0] is a column header
50          // link[1] is paging link
51  
52          // 1. User clicks column 3 header to sort ascending. User returned to page 1.
53          response = response.getLinks()[0].click();
54          Assertions.assertNotNull(response.getElementWithID("PAGEONE"), "Not in page one as expected");
55  
56          // 2. User clicks column 3 header to sort descending. User returned to page 1.
57          response = response.getLinks()[0].click();
58          Assertions.assertNotNull(response.getElementWithID("PAGEONE"), "Not in page one as expected");
59  
60          // 3. User navigates to page other than page 1.
61          response = response.getLinks()[1].click();
62          Assertions.assertNotNull(response.getElementWithID("OTHERPAGE"), "Not in page two as expected");
63  
64          // 4. User clicks column 3 header to sort ascending. User NOT returned to page 1, rather, user stays on current
65          // page number.
66          response = response.getLinks()[0].click();
67          Assertions.assertNotNull(response.getElementWithID("PAGEONE"), "Not in page one as expected");
68      }
69  
70  }
