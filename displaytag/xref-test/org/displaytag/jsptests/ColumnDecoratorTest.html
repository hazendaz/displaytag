22  package org.displaytag.jsptests;
23  
24  import org.displaytag.decorator.DateColumnDecorator;
25  import org.displaytag.test.DisplaytagCase;
26  import org.displaytag.test.KnownTypes;
27  import org.junit.Assert;
28  import org.junit.Test;
29  
30  import com.meterware.httpunit.GetMethodWebRequest;
31  import com.meterware.httpunit.WebRequest;
32  import com.meterware.httpunit.WebResponse;
33  import com.meterware.httpunit.WebTable;
34  
35  
36  /**
37   * Tests for column decorators.
38   * @author Fabrizio Giustina
39   * @version $Revision$ ($Author$)
40   */
41  public class ColumnDecoratorTest extends DisplaytagCase
42  {
43  
44      /**
45       * @see org.displaytag.test.DisplaytagCase#getJspName()
46       */
47      public String getJspName()
48      {
49          return "columndecorator.jsp";
50      }
51  
52      /**
53       * Checks that the generated page contains decorated values.
54       * @param jspName jsp name, with full path
55       * @throws Exception any axception thrown during test.
56       */
57      @Override
58      @Test
59      public void doTest() throws Exception
60      {
61          WebRequest request = new GetMethodWebRequest(getJspUrl(getJspName()));
62  
63          WebResponse response = this.runner.getResponse(request);
64  
65          if (this.log.isDebugEnabled())
66          {
67              this.log.debug("RESPONSE: " + response.getText());
68          }
69  
70          WebTable[] tables = response.getTables();
71  
72          Assert.assertEquals("Wrong number of tables.", 1, tables.length);
73  
74          Assert.assertEquals(
75              "Expected decorated value not found.",
76              new DateColumnDecorator().decorate(KnownTypes.TIME_VALUE, null, null),
77              tables[0].getCellAsText(1, 0));
78      }
79  }
