41      @Test
42      void testMainTest() throws Exception {
43          final File f = File.createTempFile("inline", "pdf");
44          final String styleSheetPath = "/org/displaytag/export/asFo_us.xsl";
45          FopExportView.transform(XslTransformerTest.XML, styleSheetPath, f);
46          final PdfReader reader = new PdfReader(f.getAbsolutePath());
47          // byte[] page = reader.getPageContent(1);
48          Assertions.assertEquals(1, reader.getNumberOfPages(), "Expected a valid pdf file with a single page");
49      }

61          XslTransformerTest.script();
62      }

64      /**
65       * Script.
66       *
67       * @throws Exception
68       *             the exception
69       */
70      public static void script() throws Exception {
71          final Source src = new StreamSource(new StringReader(XslTransformerTest.XML));
72          final String styleSheetPath = "/org/displaytag/export/asFo_us.xsl";
73          final InputStream styleSheetStream = FopExportView.class.getResourceAsStream(styleSheetPath);
74  
75          final Source xslt = new StreamSource(styleSheetStream);
76          final TransformerFactory factory = TransformerFactory.newInstance();
77          Transformer transformer;
78          try {
79              transformer = factory.newTransformer(xslt);
80          } catch (final TransformerConfigurationException e) {
81              throw new JspException("Cannot configure pdf export " + e.getMessage(), e); //$NON-NLS-1$
82          }
83          final StreamResult debugRes = new StreamResult(new StringWriter());
84          try {
85              transformer.transform(src, debugRes);
86          } catch (final TransformerException e) {
87              throw new JspException("error creating pdf output " + e.getMessage(), e); //$NON-NLS-1$
88          }
89          System.out.println(debugRes.getWriter());
90          FopExportView.transform(XslTransformerTest.XML, styleSheetPath, Path.of("/Users/andy/test.pdf").toFile());
91      }
