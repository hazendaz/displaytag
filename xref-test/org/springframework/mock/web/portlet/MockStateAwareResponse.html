17  package org.springframework.mock.web.portlet;
18  
19  import java.io.Serializable;
20  import java.util.Collections;
21  import java.util.HashMap;
22  import java.util.Iterator;
23  import java.util.LinkedHashMap;
24  import java.util.Map;
25  
26  import javax.portlet.MutableRenderParameters;
27  import javax.portlet.PortalContext;
28  import javax.portlet.PortletMode;
29  import javax.portlet.PortletModeException;
30  import javax.portlet.StateAwareResponse;
31  import javax.portlet.WindowState;
32  import javax.portlet.WindowStateException;
33  import javax.xml.namespace.QName;
34  
35  import org.junit.jupiter.api.Assertions;
36  import org.springframework.util.CollectionUtils;
37  
38  /**
39   * Mock implementation of the {@link javax.portlet.StateAwareResponse} interface.
40   *
41   * @author Juergen Hoeller
42   * @since 3.0
43   */
44  public class MockStateAwareResponse extends MockPortletResponse implements StateAwareResponse {
45  
46  	/** The window state. */
47  	private WindowState windowState;
48  
49  	/** The portlet mode. */
50  	private PortletMode portletMode;
51  
52  	/** The render parameters. */
53  	private final Map<String, String[]> renderParameters = new LinkedHashMap<String, String[]>();
54  
55  	/** The events. */
56  	private final Map<QName, Serializable> events = new HashMap<QName, Serializable>();
57  
58  
59  	/**
60  	 * Create a new MockActionResponse with a default {@link MockPortalContext}.
61  	 * @see org.springframework.mock.web.portlet.MockPortalContext
62  	 */
63  	public MockStateAwareResponse() {
64  		super();
65  	}
66  
67  	/**
68  	 * Create a new MockActionResponse.
69  	 * @param portalContext the PortalContext defining the supported
70  	 * PortletModes and WindowStates
71  	 */
72  	public MockStateAwareResponse(PortalContext portalContext) {
73  		super(portalContext);
74  	}
75  
76  
77  	@Override
78  	public void setWindowState(WindowState windowState) throws WindowStateException {
79  		if (!CollectionUtils.contains(getPortalContext().getSupportedWindowStates(), windowState)) {
80  			throw new WindowStateException("WindowState not supported", windowState);
81  		}
82  		this.windowState = windowState;
83  	}
84  
85  	@Override
86  	public WindowState getWindowState() {
87  		return this.windowState;
88  	}
89  
90  	@Override
91  	public void setPortletMode(PortletMode portletMode) throws PortletModeException {
92  		if (!CollectionUtils.contains(getPortalContext().getSupportedPortletModes(), portletMode)) {
93  			throw new PortletModeException("PortletMode not supported", portletMode);
94  		}
95  		this.portletMode = portletMode;
96  	}
97  
98  	@Override
99  	public PortletMode getPortletMode() {
100 		return this.portletMode;
101 	}
102 
103 	@Override
104 	public void setRenderParameters(Map<String, String[]> parameters) {
105 		Assertions.assertNotNull(parameters, "Parameters Map must not be null");
106 		this.renderParameters.clear();
107 		this.renderParameters.putAll(parameters);
108 	}
109 
110 	@Override
111 	public void setRenderParameter(String key, String value) {
112 		Assertions.assertNotNull(key, "Parameter key must not be null");
113 		Assertions.assertNotNull(value, "Parameter value must not be null");
114 		this.renderParameters.put(key, new String[] {value});
115 	}
116 
117 	@Override
118 	public void setRenderParameter(String key, String[] values) {
119 		Assertions.assertNotNull(key, "Parameter key must not be null");
120 		Assertions.assertNotNull(values, "Parameter values must not be null");
121 		this.renderParameters.put(key, values);
122 	}
123 
124 	/**
125 	 * Gets the render parameter.
126 	 *
127 	 * @param key the key
128 	 * @return the render parameter
129 	 */
130 	public String getRenderParameter(String key) {
131 		Assertions.assertNotNull(key, "Parameter key must not be null");
132 		String[] arr = this.renderParameters.get(key);
133 		return (arr != null && arr.length > 0 ? arr[0] : null);
134 	}
135 
136 	/**
137 	 * Gets the render parameter values.
138 	 *
139 	 * @param key the key
140 	 * @return the render parameter values
141 	 */
142 	public String[] getRenderParameterValues(String key) {
143 		Assertions.assertNotNull(key, "Parameter key must not be null");
144 		return this.renderParameters.get(key);
145 	}
146 
147 	/**
148 	 * Gets the render parameter names.
149 	 *
150 	 * @return the render parameter names
151 	 */
152 	public Iterator<String> getRenderParameterNames() {
153 		return this.renderParameters.keySet().iterator();
154 	}
155 
156 	@Override
157 	public Map<String, String[]> getRenderParameterMap() {
158 		return Collections.unmodifiableMap(this.renderParameters);
159 	}
160 
161 	@Override
162 	public void removePublicRenderParameter(String name) {
163 		this.renderParameters.remove(name);
164 	}
165 
166 	@Override
167 	public void setEvent(QName name, Serializable value) {
168 		this.events.put(name, value);
169 	}
170 
171 	@Override
172 	public void setEvent(String name, Serializable value) {
173 		this.events.put(new QName(name), value);
174 	}
175 
176 	/**
177 	 * Gets the event names.
178 	 *
179 	 * @return the event names
180 	 */
181 	public Iterator<QName> getEventNames() {
182 		return this.events.keySet().iterator();
183 	}
184 
185 	/**
186 	 * Gets the event.
187 	 *
188 	 * @param name the name
189 	 * @return the event
190 	 */
191 	public Serializable getEvent(QName name) {
192 		return this.events.get(name);
193 	}
194 
195 	/**
196 	 * Gets the event.
197 	 *
198 	 * @param name the name
199 	 * @return the event
200 	 */
201 	public Serializable getEvent(String name) {
202 		return this.events.get(new QName(name));
203 	}
204 
205   @Override
206   public MutableRenderParameters getRenderParameters() {
207     // TODO Auto-generated method stub
208     return null;
209   }
210 
211 }
