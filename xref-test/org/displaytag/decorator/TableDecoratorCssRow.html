7   package org.displaytag.decorator;
8   
9   /**
10   * Test decorator used in tests.
11   */
12  public class TableDecoratorCssRow extends TableDecorator {
13  
14      /**
15       * Adds the row id.
16       *
17       * @return the string
18       *
19       * @see org.displaytag.decorator.TableDecorator#addRowId()
20       */
21      @Override
22      public String addRowId() {
23          return "rowid" + this.getViewIndex();
24      }
25  
26      /**
27       * Adds the row class.
28       *
29       * @return the string
30       *
31       * @see org.displaytag.decorator.TableDecorator#addRowClass()
32       */
33      @Override
34      public String addRowClass() {
35          if (this.getViewIndex() == 2) {
36              return "highlighted";
37          }
38          return null;
39      }
40  
41  }
