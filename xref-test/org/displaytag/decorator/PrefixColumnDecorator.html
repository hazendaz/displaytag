7   package org.displaytag.decorator;
8   
9   import jakarta.servlet.jsp.PageContext;
10  
11  import org.displaytag.exception.DecoratorException;
12  import org.displaytag.properties.MediaTypeEnum;
13  
14  /**
15   * A column decorator which adds a "decorated: " prefix.
16   */
17  public class PrefixColumnDecorator implements DisplaytagColumnDecorator {
18  
19      /**
20       * Decorate.
21       *
22       * @param columnValue
23       *            the column value
24       * @param pageContext
25       *            the page context
26       * @param media
27       *            the media
28       *
29       * @return the object
30       *
31       * @throws DecoratorException
32       *             the decorator exception
33       *
34       * @see org.displaytag.decorator.DisplaytagColumnDecorator#decorate(java.lang.Object,
35       *      jakarta.servlet.jsp.PageContext, org.displaytag.properties.MediaTypeEnum)
36       */
37      @Override
38      public Object decorate(final Object columnValue, final PageContext pageContext, final MediaTypeEnum media)
39              throws DecoratorException {
40          return "decorated: " + columnValue;
41      }
42  
43  }
