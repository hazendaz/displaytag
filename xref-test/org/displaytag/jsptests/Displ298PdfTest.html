12  import com.meterware.httpunit.WebResponse;
13  
14  import java.io.InputStream;
15  
16  import org.displaytag.export.ExportViewFactory;
17  import org.displaytag.properties.MediaTypeEnum;
18  import org.displaytag.tags.TableTagParameters;
19  import org.displaytag.test.DisplaytagCase;
20  import org.displaytag.util.ParamEncoder;
21  import org.junit.jupiter.api.Assertions;
22  import org.junit.jupiter.api.Test;
23  
24  /**
25   * The Class Displ298PdfTest.
26   */
27  class Displ298PdfTest extends DisplaytagCase {
28  
29      /**
30       * Gets the jsp name.
31       *
32       * @return the jsp name
33       *
34       * @see org.displaytag.test.DisplaytagCase#getJspName()
35       */
36      @Override
37      public String getJspName() {
38          return "DISPL-298.jsp";
39      }
40  
41      /**
42       * Check that model modifications made by table decorator specified with in the decorator property the table tag
43       * show up in the pdf export.
44       *
45       * @throws Exception
46       *             any exception thrown during test.
47       */
48      @Override
49      @Test
50      public void doTest() throws Exception {
51  
52          final ParamEncoder encoder = new ParamEncoder("table");
53          final String mediaParameter = encoder.encodeParameterName(TableTagParameters.PARAMETER_EXPORTTYPE);
54          final WebRequest request = new GetMethodWebRequest(this.getJspUrl(this.getJspName()));
55  
56          // this will force media type initialization
57          MediaTypeEnum.registerMediaType("wpdf");
58          final ExportViewFactory factory = ExportViewFactory.getInstance();
59          factory.registerExportView("wpdf", "org.displaytag.export.DefaultPdfExportView");
60          final MediaTypeEnum pdfMedia = MediaTypeEnum.fromName("wpdf");
61          Assertions.assertNotNull(pdfMedia, "Pdf export view not correctly registered.");
62          request.setParameter(mediaParameter, Integer.toString(pdfMedia.getCode()));
63  
64          final WebResponse response = this.runner.getResponse(request);
65  
66          Assertions.assertEquals("application/pdf", response.getContentType(), "Expected a different content type.");
67  
68          final InputStream stream = response.getInputStream();
69          final byte[] result = new byte[3000];
70          stream.read(result);
71  
72          final PdfReader reader = new PdfReader(result);
73          Assertions.assertEquals(1, reader.getNumberOfPages(), "Expected a valid pdf file with a single page");
74      }
75  
76  }
