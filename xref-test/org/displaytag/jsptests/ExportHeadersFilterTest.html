7   package org.displaytag.jsptests;
8   
9   import com.meterware.httpunit.GetMethodWebRequest;
10  import com.meterware.httpunit.WebRequest;
11  import com.meterware.httpunit.WebResponse;
12  
13  import org.displaytag.properties.MediaTypeEnum;
14  import org.displaytag.tags.TableTagParameters;
15  import org.displaytag.util.ParamEncoder;
16  import org.junit.jupiter.api.Assertions;
17  import org.junit.jupiter.api.Test;
18  
19  /**
20   * Tests removal of no-cache headers.
21   */
22  class ExportHeadersFilterTest extends ExportHeadersTest {
23  
24      /**
25       * Test that headers are correctly removed.
26       *
27       * @throws Exception
28       *             any exception thrown during test.
29       */
30      @Override
31      @Test
32      public void doTest() throws Exception {
33  
34          final ParamEncoder encoder = new ParamEncoder("table");
35          final String mediaParameter = encoder.encodeParameterName(TableTagParameters.PARAMETER_EXPORTTYPE);
36  
37          // test keep
38          final WebRequest request = new GetMethodWebRequest(this.getJspUrl(this.getJspName()));
39          request.setParameter(mediaParameter, Integer.toString(MediaTypeEnum.XML.getCode()));
40  
41          // this will enable the filter!
42          request.setParameter(TableTagParameters.PARAMETER_EXPORTING, "1");
43          final WebResponse response = this.runner.getResponse(request);
44  
45          Assertions.assertNull(response.getHeaderField("Cache-Control"), "Header Cache-Control not overwritten");
46          Assertions.assertNull(response.getHeaderField("Expires"), "Header Expires not overwritten");
47          Assertions.assertNull(response.getHeaderField("Pragma"), "Header Pragma not overwritten");
48      }
49  
50  }
