7   package org.displaytag.util;
8   
9   import org.displaytag.test.URLAssertions;
10  import org.junit.jupiter.api.Test;
11  
12  /**
13   * The Class PostHrefTest.
14   */
15  class PostHrefTest {
16  
17      /**
18       * Test for URLs containing parameters.
19       */
20      @Test
21      void testHrefWithParameters() {
22          final String url = "http://www.displaytag.org/displaytag/index.jsp?param1=1&param2=2";
23          final Href href = new PostHref(new DefaultHref(url), "frm");
24          final String newUrl = href.toString();
25          URLAssertions.assertEquals("javascript:displaytagform('frm',[{f:'param1',v:'1'},{f:'param2',v:'2'}])", newUrl);
26      }
27  
28      /**
29       * Test for URLs containing parameters.
30       */
31      @Test
32      void testHrefWithParametersToBeEscaped() {
33          final String url = "http://www.displaytag.org/displaytag/index.jsp?param1=a'a&param2=2";
34          final Href href = new PostHref(new DefaultHref(url), "frm");
35          final String newUrl = href.toString();
36          URLAssertions.assertEquals("javascript:displaytagform('frm',[{f:'param1',v:'a\\'a'},{f:'param2',v:'2'}])",
37                  newUrl);
38      }
39  
40  }
