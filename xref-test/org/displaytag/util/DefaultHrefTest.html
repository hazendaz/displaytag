282     /**
283      * Testadd parameter map.
284      */
285     public void testaddParameterMap() {
286         final Map<String, String[]> parametersMap = new HashMap<>();
287 
288         parametersMap.put("modifiedArray", new String[] { "a&nbspb", "c&nbspd" });
289 
290         final DefaultHref defaultHref = new DefaultHref("");
291         defaultHref.addParameterMap(parametersMap);
292         final String[] modifiedArray = parametersMap.get("modifiedArray");
293         Assertions.assertEquals("a&nbspb", modifiedArray[0]);
294         Assertions.assertEquals("c&nbspd", modifiedArray[1]);
295     }
