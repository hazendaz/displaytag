7   package org.displaytag.util;
8   
9   import org.junit.jupiter.api.Assertions;
10  import org.junit.jupiter.api.Test;
11  
12  /**
13   * Tests for ParamEncoder.
14   */
15  class ParamEncoderTest {
16  
17      /**
18       * Test for reported DISPL-12.
19       */
20      @Test
21      void testOptionAndAnswer() {
22          Assertions.assertNotEquals(new ParamEncoder("option").encodeParameterName("x"),
23                  new ParamEncoder("answer").encodeParameterName("x"),
24                  "id 'option' and 'answer' produce the same parameter");
25      }
26  
27      /**
28       * Same parameter produce the same result.
29       */
30      @Test
31      void testEquals() {
32          Assertions.assertEquals(new ParamEncoder("equals").encodeParameterName("x"),
33                  new ParamEncoder("equals").encodeParameterName("x"), "The same parameter should produce equals key.");
34      }
35  
36      /**
37       * We don't wont param names to be too long.
38       */
39      @Test
40      void testNotTooLong() {
41          Assertions.assertTrue(
42                  new ParamEncoder("averyveryveryveryveryveryverylongidvalue").encodeParameterName("x").length() < 12);
43          Assertions.assertTrue(new ParamEncoder("test").encodeParameterName("x").length() < 12);
44          Assertions.assertTrue(new ParamEncoder("a").encodeParameterName("x").length() < 12);
45          Assertions.assertTrue(new ParamEncoder("xxxxxxxxxxxx").encodeParameterName("x").length() < 12);
46          Assertions.assertTrue(new ParamEncoder("xxxxxxxxxxxxxxxxxxxxxxxxxxxx").encodeParameterName("x").length() < 12);
47      }
48  }
