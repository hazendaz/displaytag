7   package org.displaytag.sample;

9   import java.util.List;
10  
11  /**
12   * Simple objects which holds a list.
13   */
14  public class ListHolder extends Object {
15  
16      /**
17       * contained list.
18       */
19      private List<ListObject> list;
20  
21      /**
22       * Instantiate a new ListHolder and initialize a TestList with 5 elements.
23       */
24      public ListHolder() {
25          this.list = new TestList(15, false);
26      }
27  
28      /**
29       * Returns the contained list.
30       *
31       * @return a TestList with 15 elements
32       */
33      public final List<ListObject> getList() {
34          return this.list;
35      }
36  }
